Unused struct or union member.
Bad Code 
//Bad code 
struct Foo
{
};
struct Bar
{
	 Foo f;
};
int main()
{
	 Bar b; //warning: unused variable 'b' 
}
Good Code 
//Good Code 
struct Foo
{
	 Foo() {}
};
struct Bar
{
	 Foo f;
};
int main()
{
	Bar b; //no warning! It calls Foo::Foo() into b.f
}
