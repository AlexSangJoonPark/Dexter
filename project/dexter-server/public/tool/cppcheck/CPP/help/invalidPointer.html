void func()
{
    vector* vec = new vector;
    vec->push_back(*(new string("1")));
    vec->push_back(*(new string("2")));

    for(vector::iterator itr = vec->begin(); itr != vec->end(); ++itr)
    {
        string* ptr = &(*itr);
        delete(ptr);
    }
}

void func() 
{
    vector vec;
    vec.push_back(new string("1"));
    vec.push_back(new string("2"));

    for(vector::iterator itr = vec.begin(); itr != vec.end(); ++itr)
    {
        string* ptr = *itr;
        delete(ptr);
    }
}
