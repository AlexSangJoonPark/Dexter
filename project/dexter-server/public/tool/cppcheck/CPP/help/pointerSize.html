Bad Code 
int Method1() 
{
	    int days[] = {1,2,3,4,5};
	    int *ptr = days;
	    printf("%u\n", sizeof(days));
	    printf("%u\n", sizeof(ptr));	//Wrong practice.
	    return 0;
}
#define NUM_DAYS 5
typedef int days_t[ NUM_DAYS ];
#define SIZEOF_DAYS ( sizeof( days_t ) )
	
int Method2() 
{
	    days_t  days;
	    days_t *ptr = &days;
	    printf( "SIZEOF_DAYS:  %u\n", SIZEOF_DAYS  );
	    printf( "sizeof(days): %u\n", sizeof(days) );
	    printf( "sizeof(*ptr): %u\n", sizeof(*ptr) );  //Wrong practice.
	    printf( "sizeof(ptr):  %u\n", sizeof(ptr)  );  //Wrong practice.
	    return 0;
}
