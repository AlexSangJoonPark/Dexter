Be careful when using 'auto_ptr' copy.
Bad Code 
 void f()
 {
	T* pt( new T ); 
	/*...more code...*/ 
	delete pt;
 }
Good Code 
void f()
{
	auto_ptr pt( new T );
	/*...more code...*/
} // cool: pt's destructor is called as it goes out
      // of scope, and the object is deleted automatically
